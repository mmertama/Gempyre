413template <typename T>
414std::optional<T> parse(std::string_view source, const std::optional<int>& forced_base = std::nullopt) {
415 if (source.empty())
416 return std::nullopt;
417 std::stringstream ss;
418 if (forced_base) {
419 ss << std::setbase(*forced_base) << source;
420 } else {
421 auto base = std::dec;
422 if (source.size() > 1 && source[0] == '0') {
423 if (source.size() > 2 && (source[1] == 'x' || source[1] == 'X')) {
424 base = std::hex;
425 } else {
426 base = std::oct;
427 }
428 }
429 ss << base << source;
430 }
431 T v;
432 /*static_cast<std::istream&>(ss)*/
433 ss >> v; //MSVC said it would be otherwise ambiguous
434 return !ss.fail() && (!forced_base || ss.eof()) ? std::make_optional(v) : std::nullopt;
435}

892template <class T>
893std::vector<T> slurp(std::string_view file, const size_t max = std::numeric_limits<size_t>::max()) {
894 std::vector<T> vec;
896 std::ifstream stream(std::string{file}, std::ios::in | std::ios::binary | std::ios::ate);
897 if(!stream.is_open()) {
898 log(LogLevel::Error, "Cannot open file", qq(file));
899 return vec;
900 }
901 const auto size = std::min(max, static_cast<size_t>(stream.tellg()));
902 if(size <= 0) {
903 return vec;
904 }
905 vec.resize(size / sizeof (T), 0);
906 stream.seekg(std::ios_base::beg);
907 auto ptr = reinterpret_cast<char*>(vec.data());
908 stream.read(ptr, static_cast<std::streamsize>(size));
909 return vec;
910 }
